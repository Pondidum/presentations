  public class Candidate {

    public string Name { get; private set; }
    public DateTime Dob { get; private set; }

    public Candidate(string name, DateTime dob) {
      Name = name;
      Dob = dob;
    }

    public void CorrectDateOfBirth(DateTime newDateOfBirth) {
      Dob = newDateOfBirth;
    }

    public void CorrectName(string correctedName) {
      Name = correctedName;
    }

    public void NameChangedByDeedPoll(string newName) {
      Name = newName;
    }
  }
  • Methods to change values have business-y names
  • We have to methods for changing name, with different reasoning
  • If you were doing EventSourcing, these would push different events
