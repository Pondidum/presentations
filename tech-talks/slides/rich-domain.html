  public class Candidate : AggregateRoot {

    public string Name { get; private set; }
    public DateTime Dob { get; private set; }

    public Candidate(string name, DateTime dob) {
      Apply(new CandidateCreatedEvent(Guid.NewGuid(), Name, dob));
    }

    public void CorrectName(string correctedName) {
      Apply(new NameWasCorrectedEvent(correctedName));
    }

    public void NameChangedByDeedPoll(string newName) {
      Apply(new NameWasDeedPollChangedEvent(newName));
    }

    private void Handle(CandidateCreatedEvent e) {
      ID = e.AggregateID;
      Name = e.Name;
      Dob = e.Dob;
    }

    private void Handle(NameWasCorrectedEvent e) {
      Name = e.Name;
    }

    private void Handle(NameWasDeedPollChangedEvent e) {
      Name = e.Name;
    }
  }
  • All the public methods just create an event and pass it the the apply method
  • To actually apply the event, the Handle method gets called (by magic)
  • Loading from the DB will re-run all the events on a blank candidate
  • There are optimisations to make, snapshotting.

  public class NationalInsuranceNumber {
    private readonly string _number;

    public NationalInsuranceNumber(string number) {
      Validate(number).Throw();
      _number = number;
    }

    public override string ToString() {
      return _number;
    }

    public static ValidationResult Validate(string number) {
      number = Regex.Replace(number, "\s", "");
      var prefix = number.Take(2);

      if (prefix.Count != 2 || Regex.IsMatch("^[^DFIQUV]+$", prefix)) {
        return new ValidationResult.Failure("Invalid Prefix");
      }

      //more validation

      return ValidationResult.Success();
    }
  }

  public class ValidationResult {

    public bool Passed { get; private set; }
    public string Message { get; private set; }

    public void Throw() {
      if (Passed) return;
      throw new ValidationException(this);
    }

    public static ValidationResult Success() {
      return new ValidationResult { Passed = true; };
    }

    public static ValidationResult Failure(string message) {
      return new ValidationResult { Passed = false, Message = message };
    }
  }

  public class DiscountService {
    public decimal CalculateDiscount(Basket basket) {

      var total = 0;

      if (basket.Contents.Sum(l => l.SubTotal) > 50.00M) {
        total -= basket.Contents
          .GroupBy(l => l.ProductID)
          .Where(g => g.Count >= 3)
          .Sum(g => (g.Count / 3) * g.First().Price);
      }

      return total;
    }
  }

  public class Basket {

    public IEnumerable<LineItem> Contents { get; set; }

    public Basket {
      Contents = new List<LineItem>();
    }

    public void AddProduct(Product product) {
      if (Contents.Any(l => l.ProductID = product.ID) == false) {
        ((List<LineItem>)Contents).Add(new LineItem(product));
      }

      Contents.Single(l => l.ProductID = productID).Quantity++;
    }

    public decimal CalculateTotal(DiscountService discountService) {

      var total = Contents.Sum(l => l.SubTotal);
      var discount = discountService.CalculateDiscount(this);

      return total - discount
    }
  }
